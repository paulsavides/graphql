type Building {
    id: Int
    name: String
    addressNumber: String
    street: String
}

input BuildingToCreate {
  name: String
  addressNumber: String
  street: String
}

type Position {
  id: Int
  description: String
}

input PositionToCreate {
  description: String
}

type Person {
  id: Int
  firstName: String
  lastName: String
  buildingId: Int
  positionId: Int
  building: Building
  position: Position
}

input PersonToCreate {
  firstName: String
  lastName: String
  buildingId: Int
  positionId: Int
}

type Query {
  building(id: Int): [Building]
  position(id: Int): [Position]
  person(id: Int): [Person]
}

type Mutation {
  createBuilding(input: BuildingToCreate): Building
  createPosition(input: PositionToCreate): Position
  createPerson(input: PersonToCreate): Person
}
